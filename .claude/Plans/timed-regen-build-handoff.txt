2. KIEN TRUC & QUYET DINH DA CHOT                   (#2-kien-truc)
==================================================================

2.1 ABSTRACT instance + 2 SUBCLASS (KHONG interface, KHONG policy/strategy):
  - ATimedRegenInstance (abstract, JBase): giu STATE CHUNG (count, lastTicks
    qua SavedRegenPoint, max, hardMax) + so hoc Add/Use/RaiseChange/GetCount.
  - CooldownRegenInstance: accrual +1 moi regenSeconds, neo vao moc dung cuoi.
  - DailyResetRegenInstance: reset ve resetValue tai 00:00 UTC.
  - Ly do: 2 kieu khac BAN CHAT (rolling vs moc lich) -> time-math khac han.
    Interface khong giu state (2 concrete phai copy state) -> dung abstract.
    Behavior bam chat state instance -> co-locate qua ke thua, dung tach policy.

2.2 ATimedRegenService (abstract, JBase) — logic accrual THUAN, STATELESS:
  - 3 ham: CalculatePendingRegen / AdvanceLastRegenTicks / GetSecondsUntilNext.
  - Chi Cooldown dung. Share 1 instance cho moi cooldown. De unit test.

2.3 Factory theo kind trong Manager: RegisterResource(TimedRegenConfigPoint)
  switch theo configPoint.kind -> dung dung subclass.

2.4 Overfill (di qua maxCount): tach 2 cap.
  - maxCount     = moc TIME-ACCRUAL hoi toi (regen Cooldown fill toi day).
  - hardMaxCount = TRAN TUYET DOI cho grant thu cong (Add). Cho > maxCount
    -> kieu "nang luong overfill". GetCount tra SO THAT (UI tu quyet hien thi).
  - DailyReset ÉP hardMaxCount = maxCount (KHONG overfill).
  - NormalizeHardMax: hardMaxCount < maxCount (gom 0/bo trong) -> = maxCount.
  - Khi count > maxCount: regen Cooldown TU NGU; Use keo xuong < max -> self-heal
    restart. Khong pha invariant.

2.5 Invariant Cooldown: "count < maxCount ⟺ lastRegenUtcTicks > 0".
  - Day -> tick = 0. count<max ma tick<=0 -> SELF-HEAL set tick=now.
  - DailyReset KHONG co invariant nay (tick = moc reset gan nhat).

2.6 Moc reset daily = 00:00 UTC (DateTime.UtcNow.Date.Ticks). Neu can
  local-midnight -> quyet dinh khac, hoi user.

2.7 Save schema: SavedRegenPoint (type + count + lastRegenUtcTicks) du cho ca
  2 kieu. DailyReset dung lastRegenUtcTicks lam "moc reset gan nhat".

6.3 TimedRegenService.cs

using System;
using JoyCraftSDK.Utilities;

namespace JoyCraftSDK.Game {
    /// <summary>
    /// Logic thuan tinh regen theo thoi gian thuc (UTC), generic cho moi resource.
    /// Stateless: nhan state qua tham so (last ticks, current count, config) va tra ve ket qua.
    /// Tach khoi MonoBehaviour de de unit test. Day la ban generic hoa cua HeartRegenService.
    /// </summary>
    public class TimedRegenService : ATimedRegenService {
        public override int CalculatePendingRegen(long lastRegenUtcTicks, int currentCount, int maxCount, int regenSeconds, long nowUtcTicks) {
            if (currentCount >= maxCount) return 0;
            if (lastRegenUtcTicks <= 0) return 0;
            if (regenSeconds <= 0) return 0;
            long elapsedTicks = nowUtcTicks - lastRegenUtcTicks;
            if (elapsedTicks <= 0) return 0;
            long intervalTicks = TimeSpan.FromSeconds(regenSeconds).Ticks;
            int count = (int)(elapsedTicks / intervalTicks);
            int missing = maxCount - currentCount;
            return count > missing ? missing : count;
        }

        public override long AdvanceLastRegenTicks(long lastRegenUtcTicks, int regenSeconds, int regenedCount) {
            long intervalTicks = TimeSpan.FromSeconds(regenSeconds).Ticks;
            return lastRegenUtcTicks + intervalTicks * regenedCount;
        }

        public override float GetSecondsUntilNext(long lastRegenUtcTicks, int currentCount, int maxCount, int regenSeconds, long nowUtcTicks) {
            if (currentCount >= maxCount) return 0f;
            if (regenSeconds <= 0) return 0f;
            if (lastRegenUtcTicks <= 0) return regenSeconds;
            long intervalTicks = TimeSpan.FromSeconds(regenSeconds).Ticks;
            long nextTicks = lastRegenUtcTicks + intervalTicks;
            long remainingTicks = nextTicks - nowUtcTicks;
            if (remainingTicks <= 0) return 0f;
            return (float)TimeSpan.FromTicks(remainingTicks).TotalSeconds;
        }
    }

    /// <summary>
    /// Abstract base cho timed-regen logic. Cho phep swap implementation khi can
    /// (vd: server-time based de chong cheat doi clock may).
    /// </summary>
    public abstract class ATimedRegenService : JBase {
        public abstract int CalculatePendingRegen(long lastRegenUtcTicks, int currentCount, int maxCount, int regenSeconds, long nowUtcTicks);
        public abstract long AdvanceLastRegenTicks(long lastRegenUtcTicks, int regenSeconds, int regenedCount);
        public abstract float GetSecondsUntilNext(long lastRegenUtcTicks, int currentCount, int maxCount, int regenSeconds, long nowUtcTicks);
    }
}

6.4 TimedRegenInstance.cs
------------------------------------------------------------------
using System;
using JoyCraftSDK.Game.SaveData;
using JoyCraftSDK.Utilities;
using static JoyCraftSDK.AEventManager;

namespace JoyCraftSDK.Game {
    /// <summary>
    /// Base cho runtime cua 1 resource hoi/reset theo thoi gian. Giu STATE CHUNG
    /// (count, lastRegenUtcTicks qua SavedRegenPoint) + so hoc Add/Use/RaiseChange.
    /// Phan khac ban chat giua cac kieu thoi gian (tinh pending, secondsUntilNext) la abstract.
    /// Lazy regen: chi tinh khi duoc hoi -> dung ca khi app dong/suspend.
    /// </summary>
    public abstract class ATimedRegenInstance : JBase {
        private readonly RegenResourceType type;
        private readonly int maxCount;
        private readonly int hardMaxCount;
        private readonly SavedRegenPoint savedRegenPoint;

        protected ATimedRegenInstance(RegenResourceType type, int maxCount, int hardMaxCount, SavedRegenPoint savedRegenPoint) {
            this.type = type;
            this.maxCount = maxCount;
            this.hardMaxCount = hardMaxCount;
            this.savedRegenPoint = savedRegenPoint;
        }

        public RegenResourceType Type => type;
        // maxCount: moc time-accrual hoi toi. hardMaxCount: tran tuyet doi cho grant thu cong (>= maxCount).
        public int MaxCount => maxCount;
        public int HardMaxCount => hardMaxCount;

        protected int count {
            get => savedRegenPoint.count;
            set => savedRegenPoint.count = value;
        }

        protected long lastRegenUtcTicks {
            get => savedRegenPoint.lastRegenUtcTicks;
            set => savedRegenPoint.lastRegenUtcTicks = value;
        }

        public int GetCount() {
            ApplyPendingRegen();
            return count;
        }

        /// <summary>Cong (cap o hardMaxCount — cho phep vuot maxCount neu hardMaxCount > maxCount).</summary>
        public void Add(int amount) {
            if (amount < 0) return;
            ApplyPendingRegen();
            int before = count;
            count = Math.Min(count + amount, hardMaxCount);
            int actualAdded = count - before;
            OnAfterCountChanged();
            if (actualAdded > 0) RaiseChange(actualAdded);
        }

        /// <summary>Tru khi dung. Tra false neu khong du.</summary>
        public bool Use(int amount) {
            ApplyPendingRegen();
            if (count < amount) return false;
            count -= amount;
            OnAfterCountChanged();
            RaiseChange(-amount);
            return true;
        }

        protected void RaiseChange(int changeAmount) {
            _eventBus.Raise(new OnTimedRegenChange {
                type = type,
                newAmount = count,
                changeAmount = changeAmount
            });
        }

        public abstract void ApplyPendingRegen();
        public abstract float GetSecondsUntilNext();

        /// <summary>Hook sau khi count doi do Add/Use; mac dinh no-op. Cooldown override de giu "full -> tick=0".</summary>
        protected virtual void OnAfterCountChanged() {}
    }

    /// <summary>
    /// Kieu cooldown / interval-accrual: hoi +1 moi regenSeconds, neo vao thoi diem dung cuoi,
    /// cap tai max (heart, energy). Goi invariant "count < max ⟺ tick > 0" qua self-heal.
    /// </summary>
    public class CooldownRegenInstance : ATimedRegenInstance {
        private readonly int regenSeconds;
        private readonly ATimedRegenService timedRegenService;

        public CooldownRegenInstance(RegenResourceType type, int maxCount, int hardMaxCount, int regenSeconds,
            SavedRegenPoint savedRegenPoint, ATimedRegenService timedRegenService)
            : base(type, maxCount, hardMaxCount, savedRegenPoint) {
            this.regenSeconds = regenSeconds;
            this.timedRegenService = timedRegenService;
        }

        public override void ApplyPendingRegen() {
            if (count >= MaxCount) return;

            // Self-heal: count < max ma chua co moc dem -> start countdown tu now.
            if (lastRegenUtcTicks <= 0) {
                lastRegenUtcTicks = DateTime.UtcNow.Ticks;
                return;
            }

            long now = DateTime.UtcNow.Ticks;
            int regened = timedRegenService.CalculatePendingRegen(
                lastRegenUtcTicks, count, MaxCount, regenSeconds, now);
            if (regened <= 0) return;

            count += regened;
            if (count >= MaxCount) lastRegenUtcTicks = 0;
            else lastRegenUtcTicks = timedRegenService.AdvanceLastRegenTicks(lastRegenUtcTicks, regenSeconds, regened);
            RaiseChange(regened);
        }

        public override float GetSecondsUntilNext() {
            if (count >= MaxCount) return 0f;
            return timedRegenService.GetSecondsUntilNext(
                lastRegenUtcTicks, count, MaxCount, regenSeconds, DateTime.UtcNow.Ticks);
        }

        protected override void OnAfterCountChanged() {
            if (count >= MaxCount) lastRegenUtcTicks = 0;
        }
    }

    /// <summary>
    /// Kieu daily reset: count = "so luong con dung duoc hom nay"; reset do ve resetValue
    /// khi qua moc 00:00 UTC (daily ads, daily objective). lastRegenUtcTicks = "moc reset gan nhat".
    /// </summary>
    public class DailyResetRegenInstance : ATimedRegenInstance {
        private readonly int resetValue;

        // DailyReset KHONG cho overfill -> hardMaxCount ep = maxCount.
        public DailyResetRegenInstance(RegenResourceType type, int maxCount, int resetValue,
            SavedRegenPoint savedRegenPoint)
            : base(type, maxCount, maxCount, savedRegenPoint) {
            this.resetValue = resetValue;
        }

        // DateTime.UtcNow.Date cat ve 00:00 ngay do; .Ticks la moc nua dem UTC hom nay.
        private long TodayMidnightUtcTicks => DateTime.UtcNow.Date.Ticks;

        public override void ApplyPendingRegen() {
            long todayMidnight = TodayMidnightUtcTicks;

            // Lan dau (chua co moc) -> khoi tao day theo resetValue.
            if (lastRegenUtcTicks <= 0) {
                count = resetValue;
                lastRegenUtcTicks = todayMidnight;
                RaiseChange(0);
                return;
            }

            // Da qua it nhat 1 moc nua dem ke tu lan reset cuoi -> reset.
            if (lastRegenUtcTicks < todayMidnight) {
                int before = count;
                count = resetValue;
                lastRegenUtcTicks = todayMidnight;
                if (count != before) RaiseChange(count - before);
            }
        }

        public override float GetSecondsUntilNext() {
            long nextMidnight = DateTime.UtcNow.Date.AddDays(1).Ticks;
            long remainingTicks = nextMidnight - DateTime.UtcNow.Ticks;
            return remainingTicks <= 0 ? 0f : (float)TimeSpan.FromTicks(remainingTicks).TotalSeconds;
        }
    }
}

6.5 TimedRegenManager.cs
------------------------------------------------------------------
using System.Collections.Generic;
using JoyCraftSDK.SaveSystem;
using JoyCraftSDK.Utilities;

namespace JoyCraftSDK.Game {
    /// <summary>
    /// Manager rieng quan ly cac ATimedRegenInstance (moi RegenResourceType 1 instance).
    /// Nap config tu ATimedRegenConfig (trong GameConfig) + state tu SavedRegenData.
    /// Pure C# singleton (register trong ProjectScope), route API theo type.
    /// </summary>
    public class TimedRegenManager : ATimedRegenManager {
        private readonly Dictionary<RegenResourceType, ATimedRegenInstance> instances = new();
        // Chi Cooldown can service; share 1 cai cho moi cooldown instance.
        private readonly TimedRegenService timedRegenService = new();
        private bool isInitialized;

        private SavedRegenData savedRegenData => _playerProgressData.SavedRegenData;

        // Lazy init: nap instance tu config + save o lan dung dau tien (sau khi GameData san sang).
        private void EnsureInitialized() {
            if (isInitialized) return;
            isInitialized = true;
            var timedRegenConfig = _gameConfig.timedRegenConfig;
            if (timedRegenConfig == null) return;
            foreach (var configPoint in timedRegenConfig.GetConfigPoints()) {
                RegisterResource(configPoint);
            }
        }

        // Factory theo kind: dung instance dung kieu thoi gian tu 1 config point.
        public override void RegisterResource(TimedRegenConfigPoint configPoint) {
            if (instances.ContainsKey(configPoint.type)) return;
            var savedRegenPoint = savedRegenData.GetOrCreatePoint(configPoint.type);
            ATimedRegenInstance instance = configPoint.kind switch {
                RegenKind.Cooldown => new CooldownRegenInstance(
                    configPoint.type, configPoint.maxCount, NormalizeHardMax(configPoint),
                    configPoint.regenSeconds, savedRegenPoint, timedRegenService),
                RegenKind.DailyReset => new DailyResetRegenInstance(
                    configPoint.type, configPoint.maxCount, configPoint.resetValue,
                    savedRegenPoint),
                _ => null
            };
            if (instance == null) {
                LogError($"RegenKind.None cho {configPoint.type} — khong tao duoc instance. Set kind trong TimedRegenConfig.");
                return;
            }
            instances[configPoint.type] = instance;
        }

        // hardMaxCount <= 0 hoac < maxCount -> coi nhu = maxCount (khong overfill).
        private static int NormalizeHardMax(TimedRegenConfigPoint configPoint) {
            return configPoint.hardMaxCount < configPoint.maxCount ? configPoint.maxCount : configPoint.hardMaxCount;
        }

        public override int GetCount(RegenResourceType type) {
            var instance = GetInstance(type);
            return instance == null ? 0 : instance.GetCount();
        }

        public override int GetMaxCount(RegenResourceType type) {
            var instance = GetInstance(type);
            return instance == null ? 0 : instance.MaxCount;
        }

        public override void Add(RegenResourceType type, int amount) {
            GetInstance(type)?.Add(amount);
        }

        public override bool Use(RegenResourceType type, int amount) {
            var instance = GetInstance(type);
            return instance != null && instance.Use(amount);
        }

        public override float GetSecondsUntilNext(RegenResourceType type) {
            var instance = GetInstance(type);
            return instance == null ? 0f : instance.GetSecondsUntilNext();
        }

        public override void ApplyPendingRegen(RegenResourceType type) {
            GetInstance(type)?.ApplyPendingRegen();
        }

        private ATimedRegenInstance GetInstance(RegenResourceType type) {
            EnsureInitialized();
            if (instances.TryGetValue(type, out var instance)) return instance;
            LogError($"Resource {type} chua dang ky. Them vao TimedRegenConfig hoac goi RegisterResource.");
            return null;
        }
    }

    public abstract class ATimedRegenManager : JBase {
        public abstract void RegisterResource(TimedRegenConfigPoint configPoint);
        public abstract int GetCount(RegenResourceType type);
        public abstract int GetMaxCount(RegenResourceType type);
        public abstract void Add(RegenResourceType type, int amount);
        public abstract bool Use(RegenResourceType type, int amount);
        public abstract float GetSecondsUntilNext(RegenResourceType type);
        public abstract void ApplyPendingRegen(RegenResourceType type);
    }
}

                if (configPoint.type == type) return configPoint;

            LogError($"No config for {type}. Check TimedRegenConfig SO.");
            return null;
        }
    }

    public abstract class ATimedRegenConfig : JScriptableObject {
        public abstract IReadOnlyList<TimedRegenConfigPoint> GetConfigPoints();
        public abstract TimedRegenConfigPoint GetConfigPoint(RegenResourceType type);

        public RegenKind kind;
        public int maxCount;
        public int regenSeconds;   // Cooldown: chu ky hoi 1 don vi (giay).

using System;
using System.Collections.Generic;
using JoyCraftSDK.Game;
using JoyCraftSDK.Game.SaveData;
using UnityEngine;

namespace JoyCraftSDK.SaveSystem {
    /// <summary>
    /// Save data generic cho cac resource hoi theo thoi gian.
    /// Moi resource = 1 SavedRegenPoint (type + count + last ticks).
    /// Theo pattern list cua SavedFeatureData. La sub-data cua SavedPlayerProgressData.
    /// </summary>
    [Serializable]
    public class SavedRegenData : JSaveData {
        //INCREASE THIS WHEN YOU MAKE CHANGES TO THIS CLASS
        public const int MaxVersion_SavedRegenData = 1;

        #region Data

        [SerializeField] private List<SavedRegenPoint> regenPoints = new();

        #endregion

        #region Overrides

        public override int GetMaxVersion() {
            return MaxVersion_SavedRegenData;
        }

        public override List<ISavableData> GetChildrenISavableData() {
            return null;
        }

        #endregion

        /// <summary>Lay point theo type; tao moi (count=0, ticks=0) neu chua co.</summary>
        public SavedRegenPoint GetOrCreatePoint(RegenResourceType type) {
            foreach (var regenPoint in regenPoints) {
                if (regenPoint.type == type) return regenPoint;
            }
            var newPoint = new SavedRegenPoint { type = type };
            regenPoints.Add(newPoint);
            return newPoint;
        }

        public bool HasPoint(RegenResourceType type) {
            foreach (var regenPoint in regenPoints) {
                if (regenPoint.type == type) return true;
            }
            return false;
        }
    }
}

namespace JoyCraftSDK.Game.SaveData {
    /// <summary>State duoc luu cua 1 resource hoi theo thoi gian.</summary>
    [Serializable]
    public class SavedRegenPoint {
        public RegenResourceType type;
        public int count;
        public long lastRegenUtcTicks;
    }
